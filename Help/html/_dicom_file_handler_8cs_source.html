106 private void TraverseDirectoryRecords(DicomDirectoryRecord record, List<string> dicomFiles)
107 {
108 while (record != null)
109 {
110 if (record.Contains(DicomTag.ReferencedFileID))
111 {
112 string[] fileIDs = record.GetValues<string>(DicomTag.ReferencedFileID);
113 Debug.WriteLine($"ReferencedFileID values: {string.Join(", ", fileIDs)}");
114
115 string combinedPath = Path.Combine(dicomDirBasePath!, Path.Combine(fileIDs));
116 combinedPath = Path.GetFullPath(combinedPath);
117 Debug.WriteLine($"Generated combined file path: {combinedPath}");
118
119 if (File.Exists(combinedPath))
120 {
121 Debug.WriteLine($"Adding file to queue: {combinedPath}");
122 dicomFiles.Add(combinedPath);
123 }
124 else
125 {
126 Debug.WriteLine($"File does not exist or is not accessible: {combinedPath}");
127 }
128 }
129
130 TraverseDirectoryRecords(record.LowerLevelDirectoryRecord, dicomFiles);
131 record = record.NextDirectoryRecord;
132 }

161 {
162 string tempFilePath = Path.Combine(_tempFolder, $"{Path.GetFileNameWithoutExtension(filePath)}_{Guid.NewGuid()}{Path.GetExtension(filePath)}");
163 await using (FileStream sourceStream = File.Open(filePath, FileMode.Open))
164 await using (FileStream destinationStream = File.Create(tempFilePath))

166 await sourceStream.CopyToAsync(destinationStream);

228 private async Task ProcessSingleFileAsync(string filePath, string studyInstanceUID, string newPatientID, List<string> updatedFilePaths)
229 {
230 try
231 {
232 DicomFile dicomFile = await DicomFile.OpenAsync(filePath, FileReadOption.ReadAll);
233 if (dicomFile.Dataset.GetString(DicomTag.StudyInstanceUID) == studyInstanceUID)
234 {
235 dicomFile.Dataset.AddOrUpdate(DicomTag.PatientID, newPatientID);
236 string newFilePath = Path.Combine(ModifiedFolder, Path.GetFileName(filePath));
237 await dicomFile.SaveAsync(newFilePath);
238 updatedFilePaths.Add(newFilePath);
239 }
240 }
241 catch (Exception ex)
242 {
243 Debug.WriteLine($"Error updating patient ID in file {filePath}: {ex.Message}");
244 }
245 }
