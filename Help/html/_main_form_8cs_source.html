125 private void CloseApplication()
126 {
127 Debug.WriteLine("CloseApplication method called.");
128
129 if (isSending && !confirmClose)
130 {
131 Debug.WriteLine("Transfers are in progress and confirmation is not given.");
132
133 DialogResult result = MessageBox.Show("Ci sono trasferimenti in corso. Vuoi davvero chiudere il programma?", "Trasferimenti in corso", MessageBoxButtons.YesNo, MessageBoxIcon.Warning);
134 if (result == DialogResult.Yes)
135 {
136 Debug.WriteLine("User confirmed to close the application.");
137
138 confirmClose = true;
139 MainController mainController = (MainController)this.Tag!;
140 _uiController.DisableControls();
141 dataGridView1.Enabled = false;
142 mainController.CancelSending();
143
144 Task.Delay(1000).Wait(); // Attendere per assicurarsi che i file vengano rilasciati
145 Application.Exit(); // Richiama la chiusura dell'applicazione
146 }
147 else
148 {
149 Debug.WriteLine("User canceled the close operation.");
150 }
151 }
152 else
153 {
154 Debug.WriteLine("No transfers in progress or confirmation already given.");
155 ClearTempFolder();
156 Application.Exit();
157 }

164 public static void ClearTempFolder()
165 {
166 string tempPath = Path.GetTempPath() ?? throw new InvalidOperationException("Unable to get temp path");
167 string tempDirectory = Path.Combine(tempPath, "DicomModifier");
168 if (Directory.Exists(tempDirectory))
169 {
170 DirectoryInfo di = new(tempDirectory);
171 foreach (FileInfo file in di.GetFiles())
172 {
173 file.Delete();
174 }
175 foreach (DirectoryInfo dir in di.GetDirectories())
176 {
177 dir.Delete(true);
178 }
179 }
180 }
